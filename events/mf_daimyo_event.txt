namespace = mf_daimyo

########## 参勤交代 ##########
country_event = { # 年度参勤交代要求：由 on_yearly_pulse 触发。
	id = mf_daimyo.10
	title = "mf_daimyo.10.t"
	desc = "mf_daimyo.10.d"
	picture = COUNTRY_COLLAPSE_eventPicture

	is_triggered_only = yes

	trigger = {
		sort_of_mf = yes
		NOT = { has_reform = shogunate }
		overlord = {
			OR = {
				AND = {
					has_reform = shogunate
					has_country_modifier = shogun_yuling_b
				}
				AND = {
					overlord = {
						has_reform = shogunate
						has_country_modifier = shogun_yuling_b
					}
				}
			}
		}
		NOT = {
			has_country_modifier = mf_sankin_kotai_travel
		}
	}

	immediate = {
		hidden_effect = {
			# 费用 = 发展度/10 + 与宗主首都距离/2 + 固定 3。
			export_to_variable = {
				variable_name = mf_sankin_development
				value = total_development
			}
			export_to_variable = {
				variable_name = mf_sankin_distance
				value = capital_distance
				who = overlord
			}
			set_variable = {
				which = mf_sankin_cost
				which = mf_sankin_development
			}
			divide_variable = {
				which = mf_sankin_cost
				value = 10
			}
			set_variable = {
				which = mf_sankin_distance_cost
				which = mf_sankin_distance
			}
			divide_variable = {
				which = mf_sankin_distance_cost
				value = 2
			}
			change_variable = {
				which = mf_sankin_cost
				which = mf_sankin_distance_cost
			}
			change_variable = {
				which = mf_sankin_cost
				value = 3
			}
		}
	}

	option = {
		name = "mf_daimyo.10.a"
		custom_tooltip = mf_daimyo.10.a.tt
		# 独立倾向低时才会主动接受参勤。
		trigger = {
			NOT = {
				liberty_desire = 50
			}
		}
		add_country_modifier = {
			name = mf_sankin_kotai_travel
			duration = 182
		}
		treasury_reduce = {
			variable = mf_sankin_cost
		}
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				liberty_desire = 50
			}
			modifier = {
				factor = 0
				num_of_loans = 5
			}
		}
	}

	option = {
		name = "mf_daimyo.10.c"
		custom_tooltip = mf_daimyo.10.c.tt
		# 财政困难时允许暂缓，但会付出声望、独立倾向和邻国观感代价。
		trigger = {
			num_of_loans = 5
			NOT = {
				liberty_desire = 50
			}
		}
		add_prestige = -10
		add_liberty_desire = -10
		add_country_modifier = {
			name = mf_sankin_kotai_suspended
			duration = 365
		}
		every_neighbor_country = {
			limit = {
				sort_of_mf = yes
				NOT = { has_reform = shogunate }
				is_subject_of = event_target:zhengyidajiangjun
			}
			add_opinion = {
				who = ROOT
				modifier = opinion_mf_sankin_suspended
			}
		}
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				NOT = {
					num_of_loans = 5
				}
			}
			modifier = {
				factor = 0
				liberty_desire = 50
			}
		}
	}

	option = {
		name = "mf_daimyo.10.b"
		custom_tooltip = mf_daimyo.10.b.tt
		# 高独立倾向大名会拒绝参勤，并给幕府整肃理由。
		add_country_modifier = {
			name = mf_sankin_kotai_refused
			duration = 365
		}
		set_country_flag = mf_refused_sankin_kotai
		overlord = {
			mf_change_shogun_value = {
				value = -5
			}
			add_casus_belli = {
				target = ROOT
				type = cb_disloyal_vassal
				months = 60
			}
			country_event = {
				id = mf_daimyo.11
			}
		}
		ai_chance = {
			factor = 0
			modifier = {
				factor = 100
				liberty_desire = 50
			}
		}
	}
}

province_event = { # 被恢复的大名重新纳入幕府体系。
	id = mf_daimyo.12
	title = "mf_daimyo.12.t"
	desc = "mf_daimyo.12.d"
	picture = COUNTRY_COLLAPSE_eventPicture

	is_triggered_only = yes
	hidden = yes

	immediate = {
		remove_province_modifier = mf_unlawful_daimyo_land
		owner = {
			# 归还领地可能复活已灭亡大名；若它还不在幕府下，立即重新分封。
			if = {
				limit = {
					NOT = { tag = event_target:zhengyidajiangjun }
					NOT = { is_subject_of = event_target:zhengyidajiangjun }
				}
				event_target:zhengyidajiangjun = {
					create_subject = {
						subject_type = daimyo_vassal
						subject = PREV
					}
					PREV = {
						mf_normalize_daimyo_first_layer_reform_effect = yes
					}
					mf_promote_retainer_daimyos_to_shogunate_effect = yes
				}
			}
		}
	}

	option = {
		name = "mf_daimyo.12.a"
	}
}
