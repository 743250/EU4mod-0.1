	name = Getshogun_monthly_value_PosNeg
	random = no

	text = {
		localisation_key = positive_prefix
		trigger = {
			check_variable = {
				which = shogun_monthly_value
				value = 0.01
			}
		}
	}
	text = {
		localisation_key = negative_prefix
		trigger = {
			NOT = {
				check_variable = {
					which = shogun_monthly_value
					value = 0
				}
			}
		}
	}
	text = {
		localisation_key = neutral_prefix
		trigger = {
			check_variable = {
				which = shogun_monthly_value
				value = 0
			}
			NOT = {
				check_variable = {
					which = shogun_monthly_value
					value = -0.01
				}
			}
		}
	}
	text = {
		localisation_key = neutral_prefix
		trigger = {
			always = yes
		}
	}
}
