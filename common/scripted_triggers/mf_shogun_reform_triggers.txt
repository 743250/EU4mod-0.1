########## 幕府大型计划共用判定 ##########

mf_shogun_system_holds_province = {
	# 从幕府 ROOT 视角检查单个省份是否在幕府体系内。
	OR = {
		country_or_non_sovereign_subject_holds = ROOT
		owner = {
			sort_of_mf = yes
			OR = {
				is_subject_of = ROOT
				overlord = {
					is_subject_of = ROOT
				}
			}
		}
	}
}

mf_shogun_controls_japan_region = {
	# 从幕府 ROOT 视角检查，日本地区全部由幕府或非主权属国持有。
	japan_region = {
		type = all
		OR = {
			mf_shogun_system_holds_province = yes
		}
	}
}

mf_shogun_controls_japan_region_except_hokkaido = {
	# 北海道以外的日本地区全部由幕府或非主权属国持有。
	japan_region = {
		type = all
		OR = {
			province_id = 1031
			province_id = 1032
			province_id = 1847
			province_id = 1852
			province_id = 4193
			mf_shogun_system_holds_province = yes
		}
	}
}

mf_shogun_controls_hokkaido_area = {
	# 北海道全境由幕府或非主权属国持有。
	hokkaido_area = {
		type = all
		OR = {
			mf_shogun_system_holds_province = yes
		}
	}
}

mf_all_direct_daimyos_have_no_forbidden_subjects = {
	# 直属大名只能保留本模组允许的大名层级属国。
	all_subject_country = {
		OR = {
			NOT = {
				sort_of_mf = yes
				OR = {
					is_subject_of_type = daimyo_vassal
					is_subject_of_type = mf_daimyo_vassal
				}
			}
			NOT = {
				any_subject_country = {
					NOT = {
						is_subject_of_type = mf_retainer_daimyo_vassal
					}
				}
			}
		}
	}
}
