	can_concentrate_development = no

	can_be_annexed = no # 不能被吞并

	trust_on_start = 0
	
	base_liberty_desire = 10.0

	liberty_desire_development_ratio = 0.65

	should_quit_wars_on_activation = yes

	can_fight = {

		same_overlord = daimyo_vassal

	grant_province = yes
	place_relative_on_throne = no
	enforce_religion = no
	enforce_culture = no
	seize_territory = yes

	contribute_to_capital = yes
	conscript_general = yes
	force_isolation = yes
	return_land = yes
	# 特殊幕府动作：体现战国大名体系的历史味道。
	sword_hunt = yes
	sankin_kotai = yes
	expel_ronin = yes
	
	
	modifier_overlord = clear 
	modifier_subject = clear 
	modifier_subject = {
		modifier = mf_daimyo_nation
	}
	modifier_subject = {
		modifier = tributary_state_behind_overlord_tech_adm
		trigger = {
			overlord = {
				has_reform = shogunate
			}
			any_known_country = {
				mf_daimyo_foreign_learning_range = yes
				ROOT = {
					NOT = { adm_tech = PREV }
				}
			}
		}
	}
	modifier_subject = {
		modifier = tributary_state_behind_overlord_tech_dip
		trigger = {
			overlord = {
				has_reform = shogunate
			}
			any_known_country = {
				mf_daimyo_foreign_learning_range = yes
				ROOT = {
					NOT = { dip_tech = PREV }
				}
			}
		}
	}
	modifier_subject = {
		modifier = tributary_state_behind_overlord_tech_mil
		trigger = {
			overlord = {
				has_reform = shogunate
			}
			any_known_country = {
				mf_daimyo_foreign_learning_range = yes
				ROOT = {
					NOT = { mil_tech = PREV }
				}
			}
		}
	}
	modifier_subject = {
		modifier = mf_tozama_foreign_institution_learning
		trigger = {
			overlord = {
				has_reform = shogunate
			}
			any_known_country = {
				mf_daimyo_foreign_learning_range = yes
				institution_difference = {
					who = ROOT
					value = 1
				}
			}
		}
	}
	forcelimit_to_overlord = 0.1 # 普通大名给幕府提供基础陆军上限
	naval_forcelimit_to_overlord = 0.1 # 普通大名给幕府提供基础海军上限
	modifier_subject = {
		modifier = mf_daimyo_authority_siege_restriction # 高统御限制大名攻城效率
		trigger = {
			overlord = {
				has_reform = shogunate
			}
			1020 = {
				check_variable = {
					which = shogun_value
					value = 50
				}
			}
		}
	}
}
