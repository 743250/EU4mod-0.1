	# ROOT 婵炴垶鎸搁幖顐ょ礊鐎ｎ喖绀堢€广儱鎳愰搴ㄥ箹鐎涙ê娈犻柍褜鍓欓崐鐣屸偓姘ュ灪瀵板嫰鍩￠崘鈺冩毈闁诲海顢婇崺鏍涘畝鍕€舵い鎾寸箞椤庢鏌熺€涙澧抽柛鏃€姘ㄧ划瀣嫚閼碱剦鍟傞柟鐓庣摠缁牏绱炵€ｎ剙瀵插ù鍏肩懅缁€澶愭煕閹烘挾鎳囬柛锝嗘倐瀵剚锛愭担铏圭厾闂佺儵鏅濋ˉ鎰閻楀牆绶炵憸宥夊箖閺囩喓鈻旀い鎾跺枎閻﹀綊鎮楃憴鍕惛闁?
	if = {
		limit = {
			has_saved_global_event_target = tianhuang
			event_target:tianhuang = {
				exists = no
			}
		}
		clear_global_event_target = tianhuang
	}
	if = {
		limit = {
			has_saved_global_event_target = tianhuang
			event_target:tianhuang = {
				exists = yes
				NOT = { is_subject_of = ROOT }
			}
		}
		event_target:tianhuang = {
			clr_country_flag = mf_tianhuang_target
		}
		clear_global_event_target = tianhuang
	}
	if = {
		limit = {
			NOT = { has_saved_global_event_target = tianhuang }
		}
		random_subject_country = {
			limit = {
				NOT = { has_country_flag = mf_xiayi_target }
				NOT = { has_country_flag = mf_tianhuang_target }
			}
			save_global_event_target_as = tianhuang
			set_country_flag = mf_tianhuang_target
		}
	}
}

mf_set_xiayi_campaign_target_effect = {
	# 设置虾夷征讨目标，战争借口由 cb_types 条件直接提供。
	every_country = {
		limit = { has_country_flag = mf_xiayi_target }
		clr_country_flag = mf_xiayi_target
	}
	if = {
		limit = { has_saved_global_event_target = xiayi }
		clear_global_event_target = xiayi
	}
	save_global_event_target_as = xiayi
	set_country_flag = mf_xiayi_target
}
