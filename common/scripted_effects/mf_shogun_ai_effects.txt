mf_apply_xuanxiafengxing_effect = {
	# 复位旧天皇家族目标
	if = {
		limit = { has_saved_global_event_target = tianhuang }
		event_target:tianhuang = {
			clr_country_flag = mf_tianhuang_target
			clear_global_event_target = tianhuang
		}
	}
	FROM = {
		save_global_event_target_as = tianhuang
		set_country_flag = mf_tianhuang_target
	}
	clr_country_flag = xuanxiafengxing_flag
	set_country_flag = xuanxiafengxing_flag
	mf_change_shogun_value = {
		value = -10
	}
}

mf_ai_apply_xuanxiafengxing_effect = {
	# AI 选天皇目标
	if = {
		limit = { has_saved_global_event_target = tianhuang }
		event_target:tianhuang = {
			clr_country_flag = mf_tianhuang_target
			clear_global_event_target = tianhuang
		}
	}
	random_subject_country = {
		limit = {
			NOT = { has_country_flag = mf_xiayi_target }
			NOT = { has_country_flag = mf_tianhuang_target }
		}
		save_global_event_target_as = tianhuang
		set_country_flag = mf_tianhuang_target
	}
	clr_country_flag = xuanxiafengxing_flag
	set_country_flag = xuanxiafengxing_flag
	mf_change_shogun_value = {
		value = -10
	}
}

mf_ai_apply_shouyexiayi_effect = {
	# AI 选虾夷目标
	if = {
		limit = { has_saved_global_event_target = xiayi }
		event_target:xiayi = {
			clr_country_flag = mf_xiayi_target
			clear_global_event_target = xiayi
		}
	}
	random_country = {
		limit = {
			NOT = { has_country_flag = mf_tianhuang_target }
			NOT = { has_country_flag = mf_xiayi_target }
			any_neighbor_country = {
				OR = {
					tag = event_target:zhengyidajiangjun
					is_subject_of = event_target:zhengyidajiangjun
				}
			}
		}
		save_global_event_target_as = xiayi
		set_country_flag = mf_xiayi_target
	}
	clr_country_flag = shouyexiayi_flag
	set_country_flag = shouyexiayi_flag
	mf_refresh_xiayi_campaign_effect = yes
}

mf_refresh_xiayi_campaign_effect = {
	# 重新刷新虾夷目标状态
	if = {
		limit = {
			has_saved_global_event_target = xiayi
			event_target:xiayi = {
				exists = no
			}
		}
		clear_global_event_target = xiayi
	}
	if = {
		limit = {
			has_saved_global_event_target = xiayi
			event_target:xiayi = {
				exists = yes
				OR = {
					tag = event_target:zhengyidajiangjun
					is_subject_of = event_target:zhengyidajiangjun
				}
			}
		}
		event_target:xiayi = {
			if = {
				limit = { is_subject_of = event_target:zhengyidajiangjun }
				change_subject_type = {
					subject_type = daimyo_vassal
				}
			}
			mf_normalize_daimyo_first_layer_reform_effect = yes
			clr_country_flag = mf_xiayi_target
		}
		clear_global_event_target = xiayi
	}
	if = {
		limit = {
			has_saved_global_event_target = xiayi
		}
		event_target:zhengyidajiangjun = { bl_shogun_refresh = yes }
	}
}
