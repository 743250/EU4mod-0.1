########## 通用费用处理 ##########
treasury_reduce = {
	hidden_effect = {
		while = {
			# 先按整数扣款，再把剩余小数放大到分位处理。
			limit = {
				check_variable = {
					which = $variable$
					value = 1
				}
			}
			add_treasury = -1
			subtract_variable = {
				which = $variable$
				value = 1
			}
		}
		multiply_variable = {
			which = $variable$
			value = 100
		}
		while = {
			limit = {
				check_variable = {
					which = $variable$
					value = 1
				}
			}
			add_treasury = -0.01
			subtract_variable = {
				which = $variable$
				value = 1
			}
		}
	}
}
year_treasury_reduce = {
	hidden_effect = {
		while = {
			# 先按整数扣年收入，再把剩余小数放大到分位处理。
			limit = {
				check_variable = {
					which = $variable$
					value = 1
				}
			}
			add_years_of_income = -1
			subtract_variable = {
				which = $variable$
				value = 1
			}
		}
		multiply_variable = {
			which = $variable$
			value = 100
		}
		while = {
			limit = {
				check_variable = {
					which = $variable$
					value = 1
				}
			}
			add_years_of_income = -0.01
			subtract_variable = {
				which = $variable$
				value = 1
			}
		}
	}
}

########## 御统与幕府外交效果 ##########
mf_change_shogun_value = {
	# 御统显示与缩放都读 1020；所有增减统一从这里走，防止越过 0-100。
	hidden_effect = {
		1020 = {
			change_variable = {
				which = shogun_value
				value = $value$
			}
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = shogun_value
							value = 0
						}
					}
				}
				set_variable = {
					which = shogun_value
					value = 0
				}
				set_variable = {
					which = shogun_value_Profix
					value = 0
				}
			}
			if = {
				limit = {
					check_variable = {
						which = shogun_value
						value = 100
					}
				}
				set_variable = {
					which = shogun_value
					value = 100
				}
				set_variable = {
					which = shogun_value_Profix
					value = 0
				}
			}
			set_variable = {
				which = shogun_value_display
				which = shogun_value
			}
			set_variable = {
				which = shogun_value_display_fraction
				which = shogun_value_Profix
			}
			multiply_variable = {
				which = shogun_value_display_fraction
				value = 0.01
			}
			change_variable = {
				which = shogun_value_display
				which = shogun_value_display_fraction
			}
		}
	}
}

mf_initialize_new_shogun_authority_effect = {
	if = {
		limit = {
			has_reform = shogunate
			OR = {
				NOT = { has_saved_global_event_target = zhengyidajiangjun }
				NOT = { tag = event_target:zhengyidajiangjun }
			}
		}
		save_global_event_target_as = zhengyidajiangjun
		1020 = {
			set_variable = {
				which = shogun_value
				value = 50
			}
			set_variable = {
				which = shogun_value_Profix
				value = 0
			}
		}
		set_global_flag = mf_shogun_initialized
		if = {
			limit = {
				NOT = { has_saved_global_event_target = tianhuang }
			}
			random_subject_country = {
				save_global_event_target_as = tianhuang
				set_country_flag = mf_tianhuang_target
			}
		}
		mf_refresh_tianhuang_target_effect = yes
		bl_mp_effect = yes
		bl_shogun_refresh = yes
	}
}





mf_promote_retainer_daimyos_to_shogunate_effect = {
	# 统一整理幕府属国层级。
	# 幕府不能保留陪臣大名，非幕府大名也不能直接拥有谱代大名。
	if = {
		limit = {
			has_reform = shogunate
		}
		save_global_event_target_as = zhengyidajiangjun
		every_subject_country = {
			limit = {
				is_subject_of_type = mf_retainer_daimyo_vassal
			}
			if = {
				limit = {
					OR = {
						has_dlc = "Mandate of Heaven"
						has_dlc = "Art of War"
						has_dlc = "Leviathan"
					}
				}
				change_subject_type = {
					subject_type = daimyo_vassal
				}
			}
			mf_normalize_daimyo_first_layer_reform_effect = yes
		}
		mf_refresh_tianhuang_target_effect = yes
		bl_shogun_refresh = yes
	}
	if = {
		limit = {
			NOT = {
				has_reform = shogunate
			}
		}
		save_global_event_target_as = mf_daimyo_rehome_lord
		every_subject_country = {
			limit = {
				is_subject_of_type = mf_daimyo_vassal
			}
			mf_rehome_daimyo_subjects_effect = yes
			if = {
				limit = {
					OR = {
						has_dlc = "Mandate of Heaven"
						has_dlc = "Art of War"
						has_dlc = "Leviathan"
					}
				}
				change_subject_type = {
					subject_type = mf_retainer_daimyo_vassal
				}
			}
			mf_normalize_daimyo_first_layer_reform_effect = yes
		}
		if = {
			limit = { has_saved_global_event_target = mf_daimyo_rehome_lord }
			clear_global_event_target = mf_daimyo_rehome_lord
		}
	}
}

mf_rehome_daimyo_subjects_effect = {
	# 把被收服或被重整的大名直属陪臣转挂到新上级，避免层级断裂。
	if = {
		limit = { has_saved_global_event_target = mf_daimyo_rehome_lord }
		every_subject_country = {
			limit = {
				OR = {
					is_subject_of_type = daimyo_vassal
					is_subject_of_type = mf_daimyo_vassal
					is_subject_of_type = mf_retainer_daimyo_vassal
				}
			}
			if = {
				limit = {
					is_subject = yes
				}
				overlord = {
					free_vassal = PREV
				}
			}
			event_target:mf_daimyo_rehome_lord = {
				vassalize = PREV
			}
			if = {
				limit = {
					event_target:mf_daimyo_rehome_lord = {
						has_reform = shogunate
					}
				}
				if = {
					limit = {
						OR = {
							has_dlc = "Mandate of Heaven"
							has_dlc = "Art of War"
							has_dlc = "Leviathan"
						}
					}
					change_subject_type = {
						subject_type = daimyo_vassal
					}
				}
				mf_normalize_daimyo_first_layer_reform_effect = yes
			}
			else = {
				if = {
					limit = {
						OR = {
							has_dlc = "Mandate of Heaven"
							has_dlc = "Art of War"
							has_dlc = "Leviathan"
						}
					}
					change_subject_type = {
						subject_type = mf_retainer_daimyo_vassal
					}
				}
				mf_normalize_daimyo_first_layer_reform_effect = yes
			}
		}
	}
}

mf_relink_direct_daimyos_after_shogunate_transfer_effect = {
	# 原版 japan.1 只会重挂普通大名；这里补上本 mod 新增的谱代、陪臣和异常独立大名。
	if = {
		limit = {
			has_reform = shogunate
		}
		every_country = {
			limit = {
				NOT = { has_reform = shogunate }
				NOT = { is_subject_of = ROOT }
				NOT = { war_with = ROOT }
				OR = {
					is_subject_of_type = daimyo_vassal
					is_subject_of_type = mf_daimyo_vassal
					AND = {
						is_subject_of_type = mf_retainer_daimyo_vassal
						overlord = {
							has_reform = shogunate
						}
					}
					AND = {
						has_reform = indep_daimyo
						capital_scope = { region = japan_region }
						culture_group = japanese_g
						NOT = { is_subject_of_type = mf_retainer_daimyo_vassal }
					}
					AND = {
						OR = {
							has_reform = daimyo
							has_reform = mf_daimyo
						}
						capital_scope = { region = japan_region }
						culture_group = japanese_g
						NOT = { is_subject_of_type = mf_retainer_daimyo_vassal }
					}
				}
			}
			if = {
				limit = {
					is_subject_of_type = mf_daimyo_vassal
				}
				set_country_flag = mf_relink_was_fudai_daimyo
			}
			if = {
				limit = {
					is_subject = yes
				}
				overlord = {
					free_vassal = PREV
				}
			}
			ROOT = {
				vassalize = PREV
			}
			# 新幕府转移只重挂当前对象，不拆散该大名自己的合法陪臣层级。

				limit = {
					is_subject_of = ROOT
					OR = {
						has_dlc = "Mandate of Heaven"
						has_dlc = "Art of War"
						has_dlc = "Leviathan"
					}
				}
				if = {
					limit = {
						has_country_flag = mf_relink_was_fudai_daimyo
					}
					change_subject_type = {
						subject_type = mf_daimyo_vassal
					}
				}
				else = {
					change_subject_type = {
						subject_type = daimyo_vassal
					}
				}
			}
			if = {
				limit = {
					is_subject_of = ROOT
				}
				mf_normalize_daimyo_first_layer_reform_effect = yes
			}
			clr_country_flag = mf_relink_was_fudai_daimyo
		}
		every_country = {
			limit = {
				NOT = { has_reform = shogunate }
				NOT = { is_subject_of = ROOT }
				war_with = ROOT
				OR = {
					is_subject_of_type = daimyo_vassal
					is_subject_of_type = mf_daimyo_vassal
					AND = {
						is_subject_of_type = mf_retainer_daimyo_vassal
						overlord = {
							has_reform = shogunate
						}
					}
					AND = {
						has_reform = indep_daimyo
						capital_scope = { region = japan_region }
						culture_group = japanese_g
						NOT = { is_subject_of_type = mf_retainer_daimyo_vassal }
					}
					AND = {
						OR = {
							has_reform = daimyo
							has_reform = mf_daimyo
						}
						capital_scope = { region = japan_region }
						culture_group = japanese_g
						NOT = { is_subject_of_type = mf_retainer_daimyo_vassal }
					}
				}
			}
			ROOT = {
				add_casus_belli = {
					target = PREV
					type = cb_vassalize_mission
					months = 300
				}
			}
		}
		mf_promote_retainer_daimyos_to_shogunate_effect = yes
	}
}

mf_normalize_daimyo_first_layer_reform_effect = {
	# 第一层改革只允许落到“大名体系”里，避免转封和刷新后残留错误改革。
	if = {
		limit = {
			OR = {
				has_reform = daimyo
				has_reform = mf_daimyo
				has_reform = indep_daimyo
				is_subject_of_type = daimyo_vassal
				is_subject_of_type = mf_daimyo_vassal
				is_subject_of_type = mf_retainer_daimyo_vassal
			}
		}
		change_government = monarchy
		remove_government_reform = daimyo
		remove_government_reform = indep_daimyo
		remove_government_reform = mf_daimyo
		if = {
			limit = {
				is_subject_of_type = mf_daimyo_vassal
			}
			add_government_reform = mf_daimyo
		}
		else = {
			add_government_reform = daimyo
		}
	}
}

mf_force_partition_daimyo_effect = {
	# 强制分割会一次性归还目标持有的已灭亡旧家核心，并由 mf_daimyo.12 重新纳入幕府。
	save_event_target_as = mf_partition_daimyo_target
	every_owned_province = {
		limit = {
			is_capital = no
			any_core_country = {
				NOT = { tag = event_target:mf_partition_daimyo_target }
				NOT = { tag = ROOT }
			}
		}
		random_core_country = {
			limit = {
				NOT = { tag = event_target:mf_partition_daimyo_target }
				NOT = { tag = ROOT }
			}
			PREV = {
				remove_province_modifier = mf_unlawful_daimyo_land
				cede_province = PREV
				province_event = {
					id = mf_daimyo.12
				}
			}
			add_opinion = {
				who = ROOT
				modifier = opinion_mf_partition_restored_land
			}
		}
	}
	add_country_modifier = {
		name = mf_shogunate_partitioned
		duration = 3650
	}
}

mf_refresh_shogun_leader_target_effect = {
	# ROOT 不是幕府本体时，清掉陈旧的征夷大将军目标，避免旧足利继续吃到幕府决议。
	if = {
		limit = {
			has_saved_global_event_target = zhengyidajiangjun
			event_target:zhengyidajiangjun = {
				exists = no
			}
		}
		clear_global_event_target = zhengyidajiangjun
	}
	if = {
		limit = {
			has_saved_global_event_target = zhengyidajiangjun
			event_target:zhengyidajiangjun = {
				exists = yes
				NOT = { has_reform = shogunate }
			}
		}
		clear_global_event_target = zhengyidajiangjun
	}
	if = {
		limit = {
			has_reform = shogunate
			OR = {
				NOT = { has_saved_global_event_target = zhengyidajiangjun }
				NOT = { tag = event_target:zhengyidajiangjun }
			}
		}
		save_global_event_target_as = zhengyidajiangjun
	}
}
