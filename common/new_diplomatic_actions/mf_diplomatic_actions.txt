xuanxiafengxing = { # 宣下奉行：完成“禁中并公”后，幕府可重新指定天皇目标。
	category = shogun
	alert_index = 17
	alert_tooltip = xuanxiafengxing_tooltip
	require_acceptance = no

	is_visible = {
		ROOT = {
			tag = event_target:zhengyidajiangjun
			has_global_flag = shogun_weixin.d_global_flag
		}
		FROM = {
			is_subject_of = ROOT
			NOT = { has_country_flag = mf_xiayi_target }
			NOT = { has_country_flag = mf_tianhuang_target }
		}
	}

	is_allowed = {
		custom_trigger_tooltip = {
			tooltip = mf_xuanxiafengxing_authority_tt
			1020 = {
				check_variable = {
					which = shogun_value
					value = 10
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = mf_xuanxiafengxing_cooldown_tt
			OR = {
				had_country_flag = {
					flag = xuanxiafengxing_flag
					days = 3650
				}
				NOT = { has_country_flag = xuanxiafengxing_flag }
			}
		}
	}

	on_accept = {
		if = {
			limit = { has_saved_global_event_target = tianhuang }
			event_target:tianhuang = {
				clr_country_flag = mf_tianhuang_target
				clear_global_event_target = tianhuang
			}
		}
		FROM = {
			save_global_event_target_as = tianhuang
			set_country_flag = mf_tianhuang_target
		}
		clr_country_flag = xuanxiafengxing_flag
		set_country_flag = xuanxiafengxing_flag
		mf_change_shogun_value = {
			value = -10
		}
	}

	ai_will_do = {
		always = no
	}
}
